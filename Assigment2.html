# Create a new factor variable with two levels - "weekend" and "weekday"
DayType <- function(date) {
  day <- weekdays(date,abbreviate = FALSE)
  if (day %in% c('lunes', 'martes', 'miércoles', 'jueves', 'viernes'))
      return ("weekeday")
  else if (day %in% c('sábado', 'domingo'))
      return ("weekend")
  else
       print(weekdays(date))
}
